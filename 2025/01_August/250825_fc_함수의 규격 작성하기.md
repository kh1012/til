# 첫번째 함수

```ts
// 공통 변수 묶음
type News = {
  id: number;
  time_ago: string;
  title: string;
  url: string;
  user: string;
  content: string;
};

// type intersection
type NewsFeed = News & {
  comments_count: number;
  points: number;
  read?: boolean;
};

type NewsDetail = News & {
  comments: NewsComment[];
};

type NewsComment = {
  comments: NewsComment[];
  level: number;
};
```

}
